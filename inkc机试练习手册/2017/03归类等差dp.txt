#include "pch.h"
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n;
	int dp[100][100];
	for (int i = 0; i < 100; i++)
		for (int j = 0; j < 100; j++)
			dp[i][j] = 1;
	int a[1000];
	int ma = 0,mi = 9999999;
	int len = 0;
	while (cin >> n)
	{

		for (int i = 0; i < n; i++)
		{
			cin >> a[i];
			if (ma < a[i])
				ma = a[i];
			if (mi > a[i])
				mi = a[i];


		}
		int off = ma - mi;

		for (int i = 0; i < n; i++)
		{
			for (int j = 0; j < i; j++)
			{
				int temp = a[i] - a[j];
				dp[i][temp] = max(dp[i][temp], dp[j][temp] + 1);   //tempÊÇ²î
				if (len < dp[i][temp])
					len = dp[i][temp];
			}
		}
		cout << len;

	}

	return 0;
	
}
